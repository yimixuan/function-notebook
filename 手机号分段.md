		<script type="text/javascript">
			var tel = document.getElementById("tel");
			var temp = "";
			var flag = true;
			document.documentElement.onkeyup = function(e) {
				if(tel.dataset.input && e.keyCode != 8) {
					var value = tel.value,
					arr = null;
					if (/^\D+|\D$/g.test(value)) return;//非数字开头或者结尾
					if(temp && temp == value) return;//输入框的值未发生改变
					if(flag) {//两种分割的方式
						temp = value;
						arr = value.replace(/\s/g, "").match(/(\d{3})(\d{0,4})(\d*)/); //限制输入11位号码："/(\d{3})(\d{0,4})(\d{0,4})/"
						tel.value = (arr && arr.slice(1).join(" ").replace(/\s$/,"")) || value;
					} else {
						arr = value.split(/\s?/);
						console.log(arr);
						arr[3] && arr.splice(3, 0, " ");
						arr[8] && arr.splice(8, 0, " ");
						tel.value = arr.join("");

					}
				}
			}
			tel.onfocus = function(e) {
				this.dataset.input = "true";
			}
			tel.onblur = function(e) {
				this.dataset.input = "";
			}
		</script>
