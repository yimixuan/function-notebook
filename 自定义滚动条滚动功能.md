function myAddEvent(obj,sEvent,fn){
        if(obj.attachEvent)
		    {
		        obj.attachEvent('on'+sEvent,fn);
		    }
		    else
		    {
		        obj.addEventListener(sEvent,fn,false);
		    }
}

  function onMouseWheel(ev){
	        var oEvent=ev||event;

	        var bDwon=true;    

	        bDwon=oEvent.wheelDelta?oEvent.wheelDelta < 0:oEvent.detail > 0;
	        if(bDwon)
	        {
	            setTop(scroll.offsetTop+10);
	        }
	        else
	        {
	            setTop(scroll.offsetTop-10);
	        }

	        if(oEvent.preventDefault)   
	        {
	            oEvent.preventDefault();
	        }

	        return false;
	    };

//鼠标拖动滚动条事件	
    scroll.onmousedown=function(ev){
	        var oEvent=ev||event;
	        var disY=oEvent.clientY-scroll.offsetTop;
	        document.onmousemove=function(ev)
	        {
	            var oEvent=ev||event;
	            var l=oEvent.clientY-disY;
	            setTop(l);
	        };

	        document.onmouseup=function()
	        {
	            document.onmousemove=null;
	            document.onmouseup=null;
	        };
	    };

	    {
	        if(l<0)
	        {
	            l=0;
	        }
	        else if(l>oParent.offsetHeight-scroll.offsetHeight)
	        {
	            l=oParent.offsetHeight-scroll.offsetHeight;
	        }
	        scroll.style.top=l+'px';
	        var scale=l/(oParent.offsetHeight-scroll.offsetHeight);
	        divchild.style.top=-(divchild.offsetHeight-divbox.offsetHeight)*scale+'px';
