$(function () {
	// 获取滚动条父级子级以及滚动内容区父级子级
	var s_fa=document.getElementsByClassName('s_fa')[0];
	var s_ch=document.getElementsByClassName('s_ch')[0];
	var c_fa=document.getElementsByClassName('scroll')[0];
	var c_ch=document.getElementsByClassName('scroll_left')[0];
	// 获取滚动条及滚动内容 最大滚动距离
	var max_s = s_fa.offsetHeight-s_ch.offsetHeight;
	var max_c = c_ch.offsetHeight-c_fa.offsetHeight;
	// 拖动滚动条事件
	s_ch.onmousedown = function (e) {
		var ev = e||window.event;
		var dis_y=ev.clientY - s_ch.offsetTop;
		document.onmousemove = function (e) {
			var moves=e.clientY-dis_y;
			console.log(moves+'移动')
			movefun(moves);

		};
		document.onmouseup = function (e) {
			document.onmousemove=null;
			document.onmouseup=null;
		}
	}
	// 添加滚轮事件
	objAddevent (c_fa,'mousewheel',mouseWheel);
	objAddevent (c_fa,'DOMMouseScroll',mouseWheel);
	objAddevent (s_fa,'mousewheel',mouseWheel);
	objAddevent (s_fa,'DOMMouseScroll',mouseWheel);


	function movefun(moves) {
		if (moves<=0) {
			moves=0;
			// s_ch.style.top=0;
			// c_ch.style.marginTop=0;
		}else if (moves>=max_s) {
			moves=max_s;
		}
			var scale = moves/max_s;
			s_ch.style.top=moves+'px';
			c_ch.style.marginTop=-max_c*scale+'px';
	}

	function objAddevent (obj,type,fn) {
		if (obj.addEventListener) {
			obj.addEventListener(type, fn);
		}else{
			obj.attachEvent('on'+type, fn);
		};
	}

	function mouseWheel (e) {
		var ev = e || window.event;
		var flag=true;
		flag = ev.wheelDelta?ev.wheelDelta<0:ev.detail>0;
		console.log(ev.wheelDelta+'===='+ev.detail)
		if (flag) {//页面向上滚
			movefun(s_ch.offsetTop+10);
		}else{
			movefun(s_ch.offsetTop-10);
		};
		// 阻止滚轮默认行为（页面滚动）
		if (ev.preventDefault) {
			ev.preventDefault();
		}else{
			ev.returnValue=false;
		};
		return false;
	}
	

})
